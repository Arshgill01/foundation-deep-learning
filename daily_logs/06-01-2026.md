# Code snippet I am proud of.

- import numpy as np

# 1. THE ARCHITECTURE

# Input Layer: 2 neurons (Example: Input is 0, 1)

# Hidden Layer: 2 neurons

# Output Layer: 1 neuron (Prediction)

# Sigmoid activation function and its derivative

# We use this to squash numbers between 0 and 1

def sigmoid(x):
return 1 / (1 + np.exp(-x))

def sigmoid_derivative(x):
return x \* (1 - x)

# 2. INITIALIZE DATA AND WEIGHTS

# Inputs (X) and Target Outputs (y) for XOR problem

X = np.array([[0,0], [0,1], [1,0], [1,1]])
y = np.array([[0], [1], [1], [0]])

np.random.seed(42) # For consistent results

# Weights initialized randomly

# Layer 1 weights (Input -> Hidden)

weights_input_hidden = np.random.uniform(-1, 1, (2, 2))

# Layer 2 weights (Hidden -> Output)

weights_hidden_output = np.random.uniform(-1, 1, (2, 1))

# Training settings

learning_rate = 0.5
epochs = 10000

# 3. THE TRAINING LOOP

for epoch in range(epochs):

    # --- A. FORWARD PASS ---
    # 1. Input to Hidden Layer
    hidden_layer_input = np.dot(X, weights_input_hidden)
    hidden_layer_output = sigmoid(hidden_layer_input)

    # 2. Hidden to Output Layer
    final_output_input = np.dot(hidden_layer_output, weights_hidden_output)
    predicted_output = sigmoid(final_output_input)

    # --- B. CALCULATE ERROR ---
    error = y - predicted_output

    # --- C. BACKPROPAGATION (The Chain Rule) ---

    # 1. Calculate gradient for Output Layer
    # How much did the output contribute to the error?
    d_predicted_output = error * sigmoid_derivative(predicted_output)

    # 2. Calculate gradient for Hidden Layer
    # We propagate the error backwards:
    # (Error at Output) * (Weights connecting them) * (Derivative of Hidden)
    error_hidden_layer = d_predicted_output.dot(weights_hidden_output.T)
    d_hidden_layer = error_hidden_layer * sigmoid_derivative(hidden_layer_output)

    # --- D. OPTIMIZER STEP (Update Weights) ---
    # Update weights based on how much they contributed to the error
    weights_hidden_output += hidden_layer_output.T.dot(d_predicted_output) * learning_rate
    weights_input_hidden += X.T.dot(d_hidden_layer) * learning_rate

# 4. TESTING

print(f"Final Predicted Outputs (after {epochs} epochs):\n{predicted_output}")
print("\nTarget Outputs:\n", y)
